
//pico_quack_grammar

start: statement+

statement: assignment

assignment: IDENT ":" TYPE "=" expr ";"

expr: term 
      | expr "+" term -> add
      | expr "-" term -> sub 

term: NUMBER      -> number
      | IDENT     -> var


IDENT: /[_a-zA-Z][_a-zA-Z0-9]*/
TYPE: "Int"

%import common.NEWLINE
%import common.WS
%import common.NUMBER
%ignore WS
%ignore NEWLINE
